CREATE TABLE Patient(ID INTEGER PRIMARY KEY AUTOINCREMENT,
Nom NVARCHAR(50) NOT NULL,
Prenom NVARCHAR(50) NOT NULL,
Addresse NVARCHAR(100) NOT NULL,
Tel NVARCHAR(14) NOT NULL,
Mail NVARCHAR(320) NOT NULL,
Infomed NVARCHAR(512) NOT NULL);



CREATE TABLE RENDEZVS(ID INTEGER PRIMARY KEY AUTOINCREMENT,
IDPAT INTEGER NOT NULL,
DateHeure SMALLDATETIME,
Objet NVARCHAR(512) NOT NULL,
FOREIGN KEY(ID) REFERENCES Patient(ID));
